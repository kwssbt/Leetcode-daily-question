## 线段树模板

最多添加 1e5 个元素

实现一个下标从 1 到 1e5 的区间，初始全为 0

通过区间加法，区间乘法维护区间和

用一个变量 n 记录有效元素数量，有效区间为 [ 1, n ]

`addALL(int inc)`  区间 [ 1, n ] 加 inc

`mulALL(int m)` 区间 [ 1, n ] 乘 m

`append(int val)` 有效区间扩大，新的位置赋值为 val。即 n++ ,然后区间 [ n, n ] 加 val

`getIndex(int idx)` idx++ 将下标转化为从 1 开始，超出有效区间返回 -1，否则返回 idx 处的值 即区间 [ idx ,idx ] 的和

## 代码实现 (C++)
```cpp
class Fancy {
    typedef long long ll;
    static const int MOD=1e9+7;
    static const int N=1e5+5;
    ll sum[N<<2];
    ll add[N<<2];
    ll mul[N<<2];
    int n;
    void up(int x){
        sum[x]=(sum[x<<1]+sum[x<<1|1])%MOD;
    }
    void built(int x,int l,int r){
        sum[x]=0;
        add[x]=0;
        mul[x]=1;
        if(r>l){
            int mid=(l+r)>>1;
            built(x<<1,l,mid);
            built(x<<1|1,mid+1,r);
            up(x);
        }
    }
    void lazy_add(int x,int l,int r,ll k){
        sum[x]=(sum[x]+k*(r-l+1))%MOD;
        add[x]=(add[x]+k)%MOD;
    }
    void lazy_mul(int x,ll k){
        sum[x]=sum[x]*k%MOD;
        add[x]=add[x]*k%MOD;
        mul[x]=mul[x]*k%MOD;
    }
    void down(int x,int l,int r){
        if(mul[x]!=1){
            lazy_mul(x<<1,mul[x]);
            lazy_mul(x<<1|1,mul[x]);
            mul[x]=1;
        }
        if(add[x]!=0){
            int mid=(l+r)>>1;
            lazy_add(x<<1,l,mid,add[x]);
            lazy_add(x<<1|1,mid+1,r,add[x]);
            add[x]=0;
        }
    }
    void ADD(int x,int l,int r,int ql,int qr,ll k){
        if(ql<=l&&r<=qr){
            lazy_add(x,l,r,k);
            return;
        }
        down(x,l,r);
        int mid=(l+r)>>1;
        if(ql<=mid)ADD(x<<1,l,mid,ql,qr,k);
        if(qr>mid)ADD(x<<1|1,mid+1,r,ql,qr,k);
        up(x);
    }
    void MUL(int x,int l,int r,int ql,int qr,ll k){
        if(ql<=l&&r<=qr){
            lazy_mul(x,k);
            return;
        }
        down(x,l,r);
        int mid=(l+r)>>1;
        if(ql<=mid)MUL(x<<1,l,mid,ql,qr,k);
        if(qr>mid)MUL(x<<1|1,mid+1,r,ql,qr,k);
        up(x);
    }
    ll query(int x,int l,int r,int ql,int qr){
        if(ql<=l&&r<=qr){
            return sum[x];
        }
        down(x,l,r);
        int mid=(l+r)>>1;
        ll res=0;
        if(ql<=mid)res=(res+query(x<<1,l,mid,ql,qr))%MOD;
        if(qr>mid)res=(res+query(x<<1|1,mid+1,r,ql,qr))%MOD;
        return res;
    }
public:
    Fancy() {
        n=0;
        built(1,1,100000);
    }
    
    void append(int val) {
        n++;
        ADD(1,1,100000,n,n,val);
    }
    
    void addAll(int inc) {
        if(n==0)return;
        ADD(1,1,100000,1,n,inc);
    }
    
    void multAll(int m) {
        if(n==0)return;
        MUL(1,1,100000,1,n,m);
    }
    
    int getIndex(int idx) {
        idx++;
        if(idx>n)return -1;
        return query(1,1,100000,idx,idx);
    }
};
```

- 时间复杂度: $O(Q\log N)$
- 空间复杂度: $O(N)$

## 优化

我们只需要线段树的懒更新机制

不用维护区间和，省去 `sum` 数组

单点查询


## 数学等价转换

用一个变量 $add$ 表示加法懒标记

用一个变量 $mul$ 表示乘法懒标记

容易想到每当添加新元素时结算懒标记，代价$O(N)$

这是为了使懒标记可以适配全局，从而为了这一个新元素修改所有元素

现在修改这一个新元素，使其适配当前的懒标记

值 $x$ 结算懒标记得到 $y$

$$ y = x \times mul + add $$

插入 `append(int val)`

$$ val = t \times mul + add $$

等价于插入 $t$

$$ t = \frac{val-add}{mul} $$

涉及除法取模（逆元）

## 代码实现 (C++)

```cpp
class Fancy {
    typedef long long ll;
    static const int MOD=1e9+7;
    vector<int>vals;
    ll add;
    ll mul;
    ll qpow(ll b,ll e){
        ll res=1;
        while(e){
            if(e&1)res=res*b%MOD;
            b=b*b%MOD;
            e>>=1;
        }
        return res;
    }
    ll inv(ll x){
        return qpow(x,MOD-2);
    }
public:
    Fancy() {
        this->vals.clear();
        this->add=0;
        this->mul=1;
    }
    
    void append(int val) {
        vals.push_back((val-add+MOD)*inv(mul)%MOD);
    }
    
    void addAll(int inc) {
        add=(add+inc)%MOD;
    }
    
    void multAll(int m) {
        add=add*m%MOD;
        mul=mul*m%MOD;
    }
    
    int getIndex(int idx) {
        if(idx>=vals.size()){
            return -1;
        }
        return (vals[idx]*mul%MOD+add)%MOD;
    }
};

```

- 时间复杂度: $O(N\log MOD)$
- 空间复杂度: $O(N)$
